SELECT  B.USER_ID, B.NICKNAME, B.CITY || ' ' || B.STREET_ADDRESS1 || ' ' || B.STREET_ADDRESS2 as 전체주소, SUBSTR(B.TLNO, 1, 3) || '-' || SUBSTR(B.TLNO, 4, 4) || '-' || SUBSTR(B.TLNO, 8, 4) as 전화번호
FROM    (
    SELECT WRITER_ID, COUNT(WRITER_ID) COUNTS
    FROM  USED_GOODS_BOARD
    GROUP BY WRITER_ID
) A,
        USED_GOODS_USER  B
WHERE   A.WRITER_ID = B.USER_ID
AND     A.COUNTS >= 3
ORDER BY B.USER_ID DESC;
